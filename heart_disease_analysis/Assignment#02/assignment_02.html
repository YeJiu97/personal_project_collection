# Detect outliers in numeric variables
outliers <- sapply(numeric_df, function(x) {
  lower <- quantile(x, probs = 0.25) - 1.5 * IQR(x)
  upper <- quantile(x, probs = 0.75) + 1.5 * IQR(x)
  x[x < lower | x > upper]
})
